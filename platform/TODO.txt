case WM_CLOSE:
    if (MessageBox(hwnd, L"Really quit?", L"My application", MB_OKCANCEL) == IDOK) {
        DestroyWindow(hwnd);
        internal_state.is_running = 0;
    }
    // Else: User canceled. Do nothing.
    return 0;
